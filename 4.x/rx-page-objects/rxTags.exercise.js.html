/**
 * rxTags exercises
 * @example
 * describe('default exercises', exercise.rxTags({
 *     instance: new rxTags($('.demo rx-tags')) // select one of many widgets on page
 *     sampleText: 'Tag text to use when creating and testing your tags'
 * }));
 */
function rxTags(options) {
    options = _.defaults(options, {
        sampleText: undefined,
    });
    return function () {
        var component;
        var tag;
        var numTags;
        before(function () {
            component = options.instance;
            component.count().then(function (num) {
                numTags = num;
            });
        });
        if (!_.isUndefined(options.sampleText)) {
            describe('after adding tag', function () {
                before(function () {
                    component.addTag(options.sampleText);
                    tag = component.byText(options.sampleText);
                });
                it('should have expected value', function () {
                    chai_1.expect(tag.getText()).to.eventually.equal(options.sampleText);
                });
                it('should increment total tags by 1', function () {
                    chai_1.expect(component.count()).to.eventually.equal(numTags + 1);
                });
                it('should not focus last tag', function () {
                    chai_1.expect(tag.isFocused()).to.eventually.be.false;
                });
                describe('and clicking the tag X', function () {
                    before(function () {
                        tag.remove();
                    });
                    it('should no longer exist', function () {
                        chai_1.expect(tag.isPresent()).to.eventually.be.false;
                    });
                    it('should decrement total tags by 1', function () {
                        chai_1.expect(component.count()).to.eventually.equal(numTags);
                    });
                });
            });
            describe('after adding temporary tag for removal', function () {
                before(function () {
                    component.addTag(options.sampleText);
                    tag = component.byText(options.sampleText);
                });
                describe('and typing backspace from input', function () {
                    before(function () {
                        component.sendBackspace();
                    });
                    it('should focus the last tag', function () {
                        chai_1.expect(tag.isFocused()).to.eventually.be.true;
                    });
                    describe('and typing backspace with focused tag', function () {
                        before(function () {
                            tag.backspaceRemove();
                        });
                        it('should no longer exist', function () {
                            chai_1.expect(component.byText(options.sampleText).isPresent()).to.eventually.be.false;
                        });
                        it('should decrement count by 1', function () {
                            chai_1.expect(component.count()).to.eventually.equal(numTags);
                        });
                    });
                });
            });
        }
    };
}
